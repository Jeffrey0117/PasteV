import Tesseract from 'tesseract.js';

interface OcrResult {
  text: string;
  confidence: number;
  words: Array<{
    text: string;
    bbox: { x0: number; y0: number; x1: number; y1: number };
    confidence: number;
  }>;
}

// åˆå§‹åŒ– Worker (æ‡‰ç”¨å•Ÿå‹•æ™‚)
let worker: Tesseract.Worker | null = null;

export async function initOcr(languages: string[] = ['eng', 'chi_tra']) {
  worker = await Tesseract.createWorker(languages.join('+'), 1, {
    workerPath: './node_modules/tesseract.js/dist/worker.min.js',
    langPath: './assets/tessdata',
    corePath: './node_modules/tesseract.js-core/tesseract-core.wasm.js',
  });
}

export async function recognizeImage(imageBuffer: Buffer): Promise<OcrResult> {
  if (!worker) throw new Error('OCR not initialized');

  const result = await worker.recognize(imageBuffer);

  return {
    text: result.data.text,
    confidence: result.data.confidence,
    words: result.data.words.map(w => ({
      text: w.text,
      bbox: w.bbox,
      confidence: w.confidence
    }))
  };
}

export async function terminateOcr() {
  if (worker) {
    await worker.terminate();
    worker = null;
  }
}
